convert string to num and do sum

#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>
#include <stdbool.h>

int main()
{
   FILE *file;
   file = fopen("shahaf", "w");
   fprintf(file, "1,2,3,4,5\n");
   fclose(file);
   int sum = 0;

   file = fopen("shahaf", "r");
   if (file == NULL)
   {
      perror("Error opening file for reading");
      return 1;
   }
   char read[256];

   while (fgets(read, sizeof(read), file) != NULL)
   {

      printf("%s", read);
   }
   fclose(file);
   const char delim[] = ",";
   read[strcspn(read, "\n")] = '\0';
   char *data = strtok(read, delim);
   sum = atoi(data);
   while (data != NULL)
   {

      printf("%s", data);
      data = strtok(NULL, delim);
      if (data != NULL)
         sum = sum + atoi(data);
   }
   printf("%d", sum);

   return 0;
}
